Pin D4 H2O (K2 probed at the NO port; ground shared on alligator clips for both probes leading to the 24V SMPS block screw terminal)

  - Rewriting the Python serial logger into a multi-threaded, high-speed DAQ capable of operating at a flawless 5.0 Hz (necessary to capture the rapid physical pressure spikes of millisecond ALD pulses

```
#!/usr/bin/env python3
"""
dual_logger_v5_ald.py
======================
Stable ALD Pulse Logger (5 Hz Target)
- PySerial timeout increased to 0.25s to prevent buffer desync during IGC100 housekeeping.
- Manual buffer draining added for macOS FTDI reliability.
"""

import argparse
import csv
import math
import signal
import sys
import time
import threading
from pathlib import Path

try:
    import serial
except ImportError:
    sys.exit("pyserial not installed. Run: pip3 install pyserial")

PARITY_MAP = {"N": serial.PARITY_NONE, "O": serial.PARITY_ODD, "E": serial.PARITY_EVEN}

shared_rtd = {"A_C": float("nan"), "A_ohm": float("nan"), 
              "B_C": float("nan"), "B_ohm": float("nan")}
stop_flag = False

def open_serial(port: str, baud: int, bytesize: int, parity: str, stopbits: int, timeout: float):
    s = serial.Serial(
        port=port, baudrate=baud, bytesize=bytesize,
        parity=PARITY_MAP[parity.upper()], stopbits=stopbits, timeout=timeout
    )
    time.sleep(0.1)
    return s

def _to_float(s: str) -> float:
    try:
        return float(s)
    except (ValueError, TypeError):
        return float("nan")

def rtd_worker(ls_port, baud, bits, parity, stopbits, interval):
    try:
        ls = open_serial(ls_port, baud, bits, parity, stopbits, timeout=0.5)
    except Exception as e:
        print(f"\n[RTD Thread] Could not open Lakeshore: {e}")
        return

    while not stop_flag:
        try:
            ls.reset_input_buffer()
            ls.write(b"KRDG? A\r\n")
            a_k = _to_float(ls.read_until(b'\n').decode('ascii', errors='ignore').strip())
            
            ls.write(b"SRDG? A\r\n")
            a_o = _to_float(ls.read_until(b'\n').decode('ascii', errors='ignore').strip())
            
            ls.write(b"KRDG? B\r\n")
            b_k = _to_float(ls.read_until(b'\n').decode('ascii', errors='ignore').strip())
            
            ls.write(b"SRDG? B\r\n")
            b_o = _to_float(ls.read_until(b'\n').decode('ascii', errors='ignore').strip())

            shared_rtd["A_C"] = a_k - 273.15 if not math.isnan(a_k) else float("nan")
            shared_rtd["B_C"] = b_k - 273.15 if not math.isnan(b_k) else float("nan")
            shared_rtd["A_ohm"] = a_o
            shared_rtd["B_ohm"] = b_o
        except Exception:
            pass 
        
        deadline = time.monotonic() + interval
        while time.monotonic() < deadline and not stop_flag:
            time.sleep(0.05)
            
    try:
        ls.close()
    except Exception:
        pass

def read_pressure_safe(igc, port_num: int) -> float:
    try:
        # MAC-SAFE FLUSH: Physically read out stale bytes (bypasses reset_input_buffer bug)
        while igc.in_waiting > 0:
            igc.read(igc.in_waiting)
            
        igc.write(f"GDAT?{port_num}\r".encode('ascii'))
        
        # Read until \n to ensure we capture the full \r\n terminator from the SRS
        raw = igc.read_until(b'\r').decode('ascii', errors="ignore").strip()
        
        if not raw or 'e' not in raw.lower():
            return float("nan")
        return float(raw)
    except Exception:
        return float("nan")

def main():
    global stop_flag
    ap = argparse.ArgumentParser(description="Stable ALD Pulse Logger v5")
    ap.add_argument("--ls-port",      required=True)
    ap.add_argument("--igc-port",     required=True)
    ap.add_argument("--setpoint",     type=float, required=True)
    ap.add_argument("--interval",     type=float, default=0.2, help="Pressure interval (0.2 = 5Hz target)")
    ap.add_argument("--rtd-interval", type=float, default=1.0)
    ap.add_argument("--igc-port-num", type=int,   default=4)
    ap.add_argument("--output",       default=None)

    ap.add_argument("--ls-baud",    type=int, default=9600)
    ap.add_argument("--ls-bits",    type=int, default=7)
    ap.add_argument("--ls-parity",  default="O")
    ap.add_argument("--ls-stop",    type=int, default=1)
    
    ap.add_argument("--igc-baud",   type=int, default=115200)
    ap.add_argument("--igc-bits",   type=int, default=8)
    ap.add_argument("--igc-parity", default="N")
    ap.add_argument("--igc-stop",   type=int, default=1)

    args = ap.parse_args()

    ts = time.strftime("%Y%m%d_%H%M%S")
    prefix = args.output or f"dual_ramp_{ts}"
    csv_path = Path(f"{prefix}.csv")

    print("=" * 72)
    print("  High-Speed ALD Pulse Logger  —  v5.0 (Stable 5 Hz)")
    print("=" * 72)

    try:
        # Timeout increased to 0.25s so we don't prematurely chop off a slow reply
        igc = open_serial(args.igc_port, args.igc_baud, args.igc_bits, args.igc_parity, args.igc_stop, timeout=0.25)
        igc.write(b"*IDN?\r")
        time.sleep(0.1)
        while igc.in_waiting > 0:
            igc.read(igc.in_waiting)
    except Exception as e:
        sys.exit(f"Failed to open IGC100 port: {e}")

    print("Starting Background RTD Thread...")
    t_rtd = threading.Thread(target=rtd_worker, args=(args.ls_port, args.ls_baud, args.ls_bits, args.ls_parity, args.ls_stop, args.rtd_interval), daemon=True)
    t_rtd.start()
    
    print("Benchmarking IGC100 optimized read speed...")
    t = time.monotonic(); read_pressure_safe(igc, args.igc_port_num); p_dt = time.monotonic() - t
    print(f"  Pressure read: {p_dt*1000:.0f} ms")
    if p_dt > 0: print(f"  Max theoretical pressure rate: {1/p_dt:.1f} Hz\n")

    time.sleep(1.5)
    a_c, a_o = shared_rtd["A_C"], shared_rtd["A_ohm"]
    b_c, b_o = shared_rtd["B_C"], shared_rtd["B_ohm"]
    p = read_pressure_safe(igc, args.igc_port_num)
    
    print(f"  Baseline — A: {a_c:.2f} °C   B: {b_c:.2f} °C   P: {p:.3e} Torr\n")

    csvfile = open(csv_path, "w", newline="")
    w = csv.writer(csvfile)
    w.writerow(["t_s", "A_C", "A_ohm", "B_C", "B_ohm", "BminusA_C", "P_Torr"])

    print(" Time (s)   A (°C)    B (°C)     B−A     P (Torr)")
    print("-" * 65)

    def handle_sigint(_sig, _frm):
        global stop_flag
        stop_flag = True
    signal.signal(signal.SIGINT, handle_sigint)

    t0 = time.monotonic()
    next_pressure = t0
    next_print = t0
    row_count = 0

    try:
        while not stop_flag:
            now = time.monotonic()

            if now >= next_pressure:
                p = read_pressure_safe(igc, args.igc_port_num)
                
                a_c, a_o = shared_rtd["A_C"], shared_rtd["A_ohm"]
                b_c, b_o = shared_rtd["B_C"], shared_rtd["B_ohm"]
                
                t = now - t0
                
                # Write to CSV (logs NaN as text if missed, keeps time axis perfect)
                p_str = f"{p:.4e}" if not math.isnan(p) else "NaN"
                w.writerow([f"{t:.3f}", f"{a_c:.3f}", f"{a_o:.3f}", 
                            f"{b_c:.3f}", f"{b_o:.3f}", f"{b_c-a_c:+.3f}", p_str])
                
                if not math.isnan(p):
                    row_count += 1
                
                next_pressure += args.interval
                if next_pressure < time.monotonic():
                    next_pressure = time.monotonic() + args.interval

                if now >= next_print:
                    print(f" {t:8.1f}  {a_c:7.2f}  {b_c:7.2f}  {b_c-a_c:+5.2f}  {p_str}")
                    next_print += 0.5
                    csvfile.flush()
                    
            time.sleep(0.002)

    finally:
        print("\nCtrl+C received — stopping.")
        stop_flag = True
        t_rtd.join(timeout=2.0)
        
        try: igc.close()
        except: pass
        
        csvfile.close()
        elapsed = time.monotonic() - t0
        rate = row_count / elapsed if elapsed > 0 else 0
        print("=" * 72)
        print(f"  Data points (valid) : {row_count}")
        print(f"  Total run time      : {elapsed:.1f} s")
        print(f"  Achieved rate       : {rate:.2f} Hz  <-- (Target is {1/args.interval:.1f} Hz)")
        print(f"  CSV                 : {csv_path.resolve()}")
        print("=" * 72)

if __name__ == "__main__":
    main()
```

```
/* ============================================================
   ald_commissioning_firmware.ino
   ALD Reactor — Commissioning Relay Node

   ARCHITECTURE: Python (Mac Mini) is the Primary Controller.
   Arduino is a safe Relay Driver Node.
   Python handles all timers, countdowns, and logic.
   Arduino handles hardware safety and relay execution only.

   ACTIVE-LOW relay board: LOW = relay ON = valve OPEN
                           HIGH = relay OFF = valve CLOSED

   PIN ASSIGNMENTS:
     D3 = TMA ALD valve    (LOCKED OFF — never opened in this firmware)
     D4 = H2O ALD valve    (LOCKED OFF — never opened in this firmware)
     D5 = N2 purge solenoid (SS-3J-24VDC)
     D6 = K8 main/house valve
     D7 = Physical START button (INPUT_PULLUP)
     D2 = Hardware E-STOP sense (INPUT_PULLUP)

   SERIAL COMMANDS (115200 baud, from Mac Mini Python script):
     's' or '1'  — Open N2 valve (D5 + D6 on)
     'x' or 'e' or '0'  — Close all valves (safe state)
     'r'         — Reset after E-stop (if E-stop released)

   SERIAL REPORTS (sent back to Mac Mini):
     "ACK: N2 ON"              — N2 valve opened successfully
     "ACK: N2 OFF"             — All valves closed
     "HARDWARE E-STOP"         — E-stop engaged, all safe
     "MANUAL OVERRIDE: N2 ON"  — Physical button pressed
     "MANUAL OVERRIDE: N2 OFF" — Physical button pressed again
     "RESET -> IDLE"           — Reset accepted

   SAFETY:
     - Hardware E-stop on D2 immediately safes all outputs
     - TMA and H2O pins are initialized HIGH and never touched
     - Physical button provides manual override independent of USB

   Upload with: PlatformIO → Board: Arduino Uno, Port: auto-detected
   ============================================================ */

#include <Arduino.h>

// ── Pin definitions ──────────────────────────────────────
#define VALVE_TMA_ALD    3   // LOCKED OFF
#define VALVE_H2O_ALD    4   // LOCKED OFF
#define VALVE_N2_PURGE   5   // N2 solenoid
#define PIN_HOUSE_MAIN   6   // K8 main valve
#define PIN_START_BTN    7   // Physical button (INPUT_PULLUP)
#define PIN_ESTOP_SENSE  2   // E-stop sense (INPUT_PULLUP, LOW=engaged)

// ── Relay helpers (active-LOW board) ─────────────────────
inline void relayOn (uint8_t p) { digitalWrite(p, LOW);  }
inline void relayOff(uint8_t p) { digitalWrite(p, HIGH); }

inline void safeAll() {
  relayOff(VALVE_TMA_ALD);
  relayOff(VALVE_H2O_ALD);
  relayOff(VALVE_N2_PURGE);
  relayOff(PIN_HOUSE_MAIN);
}

// ── Button debouncer ─────────────────────────────────────
struct Debounced {
  uint8_t pin;
  bool last, stable;
  unsigned long t;
};
const unsigned long DEBOUNCE_MS = 20;

bool updateDebounce(Debounced &b) {
  bool r = digitalRead(b.pin);
  if (r != b.last) { b.last = r; b.t = millis(); }
  if ((millis() - b.t) > DEBOUNCE_MS && r != b.stable) {
    b.stable = r;
    return true;
  }
  return false;
}

// ── State ─────────────────────────────────────────────────
Debounced startBtn = {PIN_START_BTN, HIGH, HIGH, 0};
bool estopPrev  = HIGH;
bool isFlowing  = false;

// ── Setup ─────────────────────────────────────────────────
void setup() {
  pinMode(VALVE_TMA_ALD,   OUTPUT); relayOff(VALVE_TMA_ALD);
  pinMode(VALVE_H2O_ALD,   OUTPUT); relayOff(VALVE_H2O_ALD);
  pinMode(VALVE_N2_PURGE,  OUTPUT); relayOff(VALVE_N2_PURGE);
  pinMode(PIN_HOUSE_MAIN,  OUTPUT); relayOff(PIN_HOUSE_MAIN);

  pinMode(PIN_ESTOP_SENSE, INPUT_PULLUP);
  pinMode(PIN_START_BTN,   INPUT_PULLUP);

  estopPrev = digitalRead(PIN_ESTOP_SENSE);

  Serial.begin(115200);
  delay(100);

  if (estopPrev == LOW) {
    Serial.println(F("HARDWARE E-STOP on boot -> all safe"));
  } else {
    Serial.println(F("=== ALD RELAY NODE READY ==="));
    Serial.println(F("Commands: s=N2 ON  x=N2 OFF  r=reset"));
  }
}

// ── Main loop ─────────────────────────────────────────────
void loop() {

  // 1. Hardware E-stop — highest priority
  bool estop = digitalRead(PIN_ESTOP_SENSE);
  if (estop == LOW && estopPrev == HIGH) {
    safeAll();
    isFlowing = false;
    Serial.println(F("HARDWARE E-STOP — all outputs safe"));
    Serial.println(F("Release E-stop then send 'r' to reset"));
  }
  if (estop == HIGH && estopPrev == LOW) {
    Serial.println(F("E-stop released — send 'r' to reset"));
  }
  estopPrev = estop;

  // 2. Physical button override (independent of USB/Python)
  if (updateDebounce(startBtn) && startBtn.stable == LOW
      && estop == HIGH) {
    if (!isFlowing) {
      relayOn(PIN_HOUSE_MAIN);
      delay(50);
      relayOn(VALVE_N2_PURGE);
      isFlowing = true;
      Serial.println(F("MANUAL OVERRIDE: N2 ON"));
    } else {
      safeAll();
      isFlowing = false;
      Serial.println(F("MANUAL OVERRIDE: N2 OFF"));
    }
  }

  // 3. Serial commands from Mac Mini
  if (Serial.available() && estop == HIGH) {
    char c = (char)Serial.read();

    if (c == 's' || c == '1') {
      relayOn(PIN_HOUSE_MAIN);
      delay(50);
      relayOn(VALVE_N2_PURGE);
      isFlowing = true;
      Serial.println(F("ACK: N2 ON"));
    }
    else if (c == 'x' || c == 'e' || c == '0') {
      safeAll();
      isFlowing = false;
      Serial.println(F("ACK: N2 OFF"));
    }
    else if (c == 'r') {
      if (estop == HIGH) {
        safeAll();
        isFlowing = false;
        Serial.println(F("RESET -> IDLE"));
      } else {
        Serial.println(F("RESET ignored: E-stop still engaged"));
      }
    }
  }
}
```

## CuO trial
Running N₂ to grow copper oxide on copper wire and copper crimps will validate the electric solenoid valve, Arduino relay control, metering valve and gas delivery all in one run.
- Full software stack validated — Arduino, IGC100 logging, Lakeshore 331 logging (until RTD failed), run_thermal_oxidation.py, test_gas_flow.py all ran end-to-end
- Oxidation achieved on copper wire + ferrule after ~243 min at ~400°C with N₂ flow and oxygen backflow — uniform dark film; phase unconfirmed.
- Run duration was not controlled — intended 90 min soak, actual ~243 min due to cap blowoff, multiple foil baffle iterations, and manual monitoring after RTD loss; conditions not tightly controlled but oxidation likely more complete as a result.
- Viton O-ring safety confirmed — QC fittings nearest to vac assembly side measured at 77°C ±10°C at ~487°C internal peak; >120°C margin below Viton 200°C limit.
- Exhaust end ran fully open — no QC on that side; foil snowball baffle only
- RTD (NB-PTCO-172) destroyed at ~58 min — bare chip sensor, no sheath; 600°C datasheet rating is for the Pt element only, not radiant furnace exposure; physically destroyed by direct radiant heat during overshoot
- Kapton wiring degraded — white/chalky from tube center toward vacuum entrance; hot zone section needs replacement
- Gas flow required multiple iterations — initial setting caused cap blowoff; finalized at 8 graduations on SS-4MG-MH ≈ 30–40 mL/min
